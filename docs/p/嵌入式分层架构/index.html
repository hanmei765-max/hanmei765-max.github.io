// driver/gpio/drv_gpio.c
#include "drv_gpio.h"
#include <stdint.h>

// ========== 寄存器映射 ==========
// STM32H7寄存器基地址定义
#define GPIOA_BASE    0x50020000
#define GPIOB_BASE    0x50020400
#define GPIOC_BASE    0x50020800

typedef struct {
    volatile uint32_t MODER;      // 0x00 - 模式寄存器
    volatile uint32_t OTYPER;     // 0x04 - 输出类型寄存器
    volatile uint32_t OSPEEDR;    // 0x08 - 输出速度寄存器
    volatile uint32_t PUPDR;      // 0x0C - 上拉/下拉寄存器
    volatile uint32_t IDR;        // 0x10 - 输入数据寄存器
    volatile uint32_t ODR;        // 0x14 - 输出数据寄存器
    volatile uint32_t BSRR;       // 0x18 - 置位/复位寄存器
    volatile uint32_t LCKR;       // 0x1C - 配置锁定寄存器
    volatile uint32_t AFR[2];     // 0x20/0x24 -  Alternate Function
} GPIO_Regs_t;

// ========== 寄存器基地址定义 ==========
#define GPIOA_REG   ((GPIO_Regs_t *)GPIOA_BASE)
#define GPIOB_REG   ((GPIO_Regs_t *)GPIOB_BASE)
#define GPIOC_REG   ((GPIO_Regs_t *)GPIOC_BASE)

// ========== 底层实现 ==========

// GPIO时钟使能（直接操作RCC寄存器）
void DRV_GPIO_EnableClock(DRV_GPIO_Port_t port) {
    // STM32H7 RCC_BASE = 0x40021000
    volatile uint32_t *rcc_cr = (uint32_t *)0x40021028;  // RCC_AHB1ENR
    
    uint32_t clock_mask = 0x0;
    switch(port) {
        case DRV_GPIO_PORT_A: clock_mask = (1 << 0);  break;
        case DRV_GPIO_PORT_B: clock_mask = (1 << 1);  break;
        case DRV_GPIO_PORT_C: clock_mask = (1 << 2);  break;
        case DRV_GPIO_PORT_D: clock_mask = (1 << 3);  break;
        case DRV_GPIO_PORT_E: clock_mask = (1 << 4);  break;
    }
    *rcc_cr |= clock_mask;
}

// GPIO初始化（驱动层最底层操作）
void DRV_GPIO_Init(DRV_GPIO_Port_t port, uint32_t pin, DRV_GPIO_Mode_t mode, 
                   DRV_GPIO_Speed_t speed, DRV_GPIO_Pull_t pull) {
    GPIO_Regs_t *gpio = NULL;
    switch(port) {
        case DRV_GPIO_PORT_A: gpio = GPIOA_REG;  break;
        case DRV_GPIO_PORT_B: gpio = GPIOB_REG;  break;
        case DRV_GPIO_PORT_C: gpio = GPIOC_REG;  break;
        default: return;
    }
    
    // 计算引脚偏移
    uint32_t pin_idx = pin / 16;
    uint32_t pin_shift = (pin % 16) * 2;
    
    // 配置模式寄存器 (MODER)
    gpio->MODER &= ~(0x3 << pin_shift);
    gpio->MODER |= (mode << pin_shift);
    
    // 配置输出速度 (OSPEEDR)
    gpio->OSPEEDR &= ~(0x3 << pin_shift);
    gpio->OSPEEDR |= (speed << pin_shift);
    
    // 配置上拉/下拉 (PUPDR)
    gpio->PUPDR &= ~(0x3 << pin_shift);
    gpio->PUPDR |= (pull << pin_shift);
}

// GPIO写操作（直接操作ODR寄存器）
void DRV_GPIO_Write(DRV_GPIO_Port_t port, uint32_t pin, uint8_t value) {
    GPIO_Regs_t *gpio = NULL;
    switch(port) {
        case DRV_GPIO_PORT_A: gpio = GPIOA_REG;  break;
        case DRV_GPIO_PORT_B: gpio = GPIOB_REG;  break;
        default: return;
    }
    
    if(value) {
        gpio->BSRR = (1 << pin);  // 置位
    } else {
        gpio->BSRR = (1 << (pin + 16));  // 复位
    }
}

// GPIO读操作（直接读IDR寄存器）
uint8_t DRV_GPIO_Read(DRV_GPIO_Port_t port, uint32_t pin) {
    GPIO_Regs_t *gpio = NULL;
    switch(port) {
        case DRV_GPIO_PORT_A: gpio = GPIOA_REG;  break;
        default: return 0;
    }
    return (gpio->IDR >> pin) & 0x1;
}

// GPIO翻转（直接写BSRR寄存器）
void DRV_GPIO_Toggle(DRV_GPIO_Port_t port, uint32_t pin) {
    GPIO_Regs_t *gpio = NULL;
    switch(port) {
        case DRV_GPIO_PORT_A: gpio = GPIOA_REG;  break;
        default: return;
    }
    gpio->BSRR = (1 << pin);  // 翻转操作
}

#ifndef HAL_GPIO_H

// HAL层GPIO状态定义
typedef enum {
    HAL_GPIO_LEVEL_LOW  = 0,
    HAL_GPIO_LEVEL_HIGH

// HAL层GPIO初始化结构
typedef struct {

    uint8_t  pull;       // 上拉/下拉

// GPIO初始化

// 引脚写操作
void HAL_GPIO_WritePin(GPIO_TypeDef *port, uint16_t pin, uint8_t level);

// 引脚读操作
uint8_t HAL_GPIO_ReadPin(GPIO_TypeDef *port, uint16_t pin);

// 引脚翻转
void HAL_GPIO_TogglePin(GPIO_TypeDef *port, uint16_t pin);

// 引脚配置
void HAL_GPIO_ConfigPin(uint8_t port, uint16_t pin, uint8_t mode);

#endif

// hal/gpio/hal_gpio.c
#include "hal_gpio.h"
#include "bsp_config.h"
#include "drv_gpio.h"

// HAL GPIO初始化
void HAL_GPIO_Init(HAL_GPIO_Init_t *init) {
    DRV_GPIO_Port_t port = DRV_GPIO_PORT_A + init->port;
    DRV_GPIO_Mode_t mode;
    DRV_GPIO_Speed_t speed = DRV_GPIO_SPEED_HIGH;
    DRV_GPIO_Pull_t pull = DRV_GPIO_NOPULL;
    
    // 模式转换
    switch(init->mode) {
        case HAL_GPIO_MODE_INPUT:       mode = DRV_GPIO_MODE_INPUT;       break;
        case HAL_GPIO_MODE_OUTPUT:      mode = DRV_GPIO_MODE_OUTPUT;      break;
        case HAL_GPIO_MODE_ALTERNATE:   mode = DRV_GPIO_MODE_ALTERNATE;   break;
        case HAL_GPIO_MODE_ANALOG:      mode = DRV_GPIO_MODE_ANALOG;      break;
        default:                        mode = DRV_GPIO_MODE_INPUT;       break;
    }
    
    // 调用驱动层接口
    DRV_GPIO_Init(port, init->pin, mode, speed, pull);
}

// HAL GPIO写操作
void HAL_GPIO_WritePin(GPIO_TypeDef *port, uint16_t pin, uint8_t level) {
    DRV_GPIO_Port_t drv_port;
    uint32_t drv_pin = pin;
    
    // 端口号映射
    if(port == GPIOA) drv_port = DRV_GPIO_PORT_A;
    else if(port == GPIOB) drv_port = DRV_GPIO_PORT_B;
    else if(port == GPIOC) drv_port = DRV_GPIO_PORT_C;
    else return;
    
    DRV_GPIO_Write(drv_port, drv_pin, level);
}

// HAL GPIO读操作
uint8_t HAL_GPIO_ReadPin(GPIO_TypeDef *port, uint16_t pin) {
    DRV_GPIO_Port_t drv_port;
    uint32_t drv_pin = pin;
    
    if(port == GPIOA) drv_port = DRV_GPIO_PORT_A;
    else if(port == GPIOB) drv_port = DRV_GPIO_PORT_B;
    else if(port == GPIOC) drv_port = DRV_GPIO_PORT_C;
    else return 0;
    
    return DRV_GPIO_Read(drv_port, drv_pin);
}

六、完整调用示例 

// 应用层：智能门锁开门逻辑
// app/door_logic.c

#include "hal_gpio.h"
#include "hal_uart.h"
#include "bsp_config.h"
#include "hal_delay.h"

// 门锁状态：0=锁定，1=解锁
uint8_t door_status = 0;

// 开门函数
void Door_Unlock(void) {
    // 使用HAL层接口（应用层不关心底层硬件）
    HAL_GPIO_WritePin(GPIOC, BSP_KEY_UNLOCK_PIN, HAL_GPIO_LEVEL_HIGH);
    
    // 延时等待解锁
    HAL_Delay(500);
    
    // 发送开门成功消息
    uint8_t msg[] = "DOOR_UNLOCK_SUCCESS\r\n";
    HAL_UART_Transmit(&huart1, msg, sizeof(msg) - 1, 100);
    
    // 设置门锁状态
    door_status = 1;
    
    // 关闭开门信号
    HAL_GPIO_WritePin(GPIOC, BSP_KEY_UNLOCK_PIN, HAL_GPIO_LEVEL_LOW);
}

// 门锁检测
void Door_Check(void) {
    uint8_t key_state = HAL_GPIO_ReadPin(GPIOC, BSP_KEY_LOCK_PIN);
    if(key_state == HAL_GPIO_LEVEL_HIGH) {
        Door_Unlock();
    }
}

七、各层文件功能对比

层级 文件类型 典型文件名 职责 稳定性
驱动层 C/H drv_gpio.c, drv_uart.h 直接操作硬件寄存器 低（依赖具体MCU）
BSP层 C/H bsp_config.h, bsp_init.c 板卡特定配置 中（依赖具体板卡）
HAL层 C/H hal_gpio.c, hal_uart.h 统一抽象接口 高（向上层稳定）
应用层 C/H door_logic.c, key_manager.c 业务逻辑 最高

八、更换MCU时的影响范围

需要修改的部分 影响层级 修改量
更换STM32型号 驱动层 修改寄存器定义
更换开发板 BSP层 修改引脚映射
更换MCU但引脚相同 HAL层 只需重编译
应用层代码 应用层 无需修改

这种分层设计使得应用层代码可以完全复用，只需修改底层三层即可适配新硬件。
